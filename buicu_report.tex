% ── Abstract ─────────────────────────────────────────────────────────
\begin{abstract}
\noindent We present BUICU, a Bayesian forecasting system for intensive care unit crowding.
Using conjugate Gamma--Poisson inference, BUICU maintains a continuously updated belief over
the patient arrival rate $\lambda$ and propagates posterior uncertainty through Monte Carlo
simulation to produce calibrated occupancy forecasts. On 180 days of synthetic
MIMIC-IV--calibrated data, the windowed model outscores the stationary model by 39.4
log-predictive units. A Law-of-Total-Variance decomposition reveals that 99.4\% of
forecast variance is irreducible Poisson noise, while a sensitivity analysis shows that
bed capacity---not arrival-rate estimation---is the dominant driver of crowding risk.
We deploy the model as an interactive Streamlit application and systematically audit five
failure modes, compounding to a $\times 2.27$ confidence-interval widening factor.
\end{abstract}

% ── 1. Introduction ──────────────────────────────────────────────────
\section{Introduction}

A single surge can push an ICU from comfortable to catastrophic overnight,
with consequences measured in delayed surgeries, hallway beds, and lives.
Deterministic census projections ignore the inherent randomness of patient
arrivals; what is needed is a probabilistic forecast that is \emph{honest
about what it does not know}.

BUICU (Belief Updating for ICU Crowding Under Uncertainty) addresses three
questions: (1)~What is the current arrival rate $\lambda$, and how
confident are we? (2)~Given the current census, what is the probability of
exceeding capacity within 48 hours? (3)~Which modelling assumptions
matter most, and where does the model break?

% ── 2. Methods ───────────────────────────────────────────────────────
\section{Methods}

\subsection{Probabilistic Model}

Let $k_t$ denote the number of admissions on day $t$. We model
each count as $N_t \mid \lambda \sim \mathrm{Poisson}(\lambda)$
and place a conjugate Gamma prior
$\lambda \sim \mathrm{Gamma}(\alpha_0, \beta_0)$ with
$\alpha_0 = 2$, $\beta_0 = 0.2$ (prior mean 10, SD $\approx 7.1$).
After $T$ days the posterior is exact:
\begin{equation}\label{eq:posterior}
    \lambda \mid k_{1:T} \;\sim\; \mathrm{Gamma}\!\left(
        \alpha_0 + \textstyle\sum k_t,\;\;
        \beta_0 + T
    \right).
\end{equation}

\noindent Each morning's update reduces to
$\alpha \leftarrow \alpha + k_t$, $\beta \leftarrow \beta + 1$---no MCMC required.

\subsection{Posterior Predictive Distribution}

Integrating out $\lambda$ yields a closed-form
\textbf{Negative Binomial} predictive:
\begin{equation}\label{eq:predpred}
    N_{\text{future}} \;\sim\; \mathrm{NegBin}\!\left(\alpha,\; \frac{\beta}{\beta + \Delta t}\right),
\end{equation}
which is overdispersed relative to Poisson, honestly reflecting both
Poisson noise and parameter uncertainty. After 180 days, the 95\% predictive
interval is $[8, 16]$ admissions.

\subsection{Windowed Model}

The stationary model treats all history equally, averaging across
regime changes. A \textbf{windowed} variant restricts sufficient statistics
to the last $W = 14$ days:
$\alpha_t^{(W)} = \alpha_0 + \sum_{s=t-W+1}^{t} k_s$,
$\beta_t^{(W)} = \beta_0 + W$.
Model comparison uses the one-step-ahead log predictive score, a
proper scoring rule that cannot be gamed by miscalibrated predictions.

\subsection{Monte Carlo Occupancy Forecasting}

Occupancy $O_t = \sum_i \mathbf{1}[a_i \le t < a_i + L_i]$ depends on
random arrivals and random lengths of stay (LOS), precluding analytic
solutions. For each of $S = 2{,}000$ trajectories we:
(i) sample $\lambda^{(s)}$ from the posterior,
(ii) draw arrivals from $\mathrm{Poisson}(\lambda^{(s)} \Delta t)$,
(iii) bootstrap LOS from the empirical distribution, and
(iv) overlay existing patients to compute $O_t^{(s)}$ over 48 hours.

% ── 3. Results ───────────────────────────────────────────────────────
\section{Results}

Results are derived from 180 days of synthetic MIMIC-IV--calibrated
data with two surge windows (days 30--50 and 110--130, rate $1.8\times$ baseline).%
\footnote{Synthetic data---calibrated to published MIMIC-IV statistics---is
used for ethical reasons; real records contain protected health information.}

\subsection{Arrival-Rate Estimation}

The posterior converges to $\hat\lambda = 11.72$ admissions/day
(95\% CI $[11.23,\, 12.23]$). Three priors---uninformative, weakly
informative, and deliberately wrong ($\mu=5$)---all converge to
within $10^{-3}$ nats KL after 180 days, confirming Bernstein--von~Mises
behaviour.

\subsection{Model Comparison}

The windowed model outscores stationary by \textbf{39.4 log-score
units}; the gap is entirely attributable to surge periods.

\subsection{Uncertainty Decomposition}

By the Law of Total Variance:
\[
    \underbrace{\frac{\Delta t / \beta}{1 + \Delta t/\beta}}_{\text{epistemic fraction}}
    \;\approx\; \frac{1/180}{1 + 1/180}
    \;\approx\; 0.6\%.
\]
\textbf{99.4\% of forecast uncertainty is irreducible Poisson noise.}
No additional data can reduce it; only process-level interventions
(e.g., triage protocols) can narrow the predictive interval.

\subsection{Crowding Forecast and Sensitivity}

Forecasting from day~36 (38/50 beds occupied) yields a 48-hour
$P(\text{overcrowded}) = \mathbf{41.5\%}$ (bootstrap 95\% CI: 40--43\%).
Table~\ref{tab:sensitivity} summarises sensitivity to key assumptions.

\subsection{Anomaly Detection}

Posterior predictive $p$-values flag 17 of 180 days as anomalous
($p < 0.025$ or $p > 0.975$); 94\% of flagged days fall within the
two surge windows.

% ── 4. Discussion ────────────────────────────────────────────────────
\section{Discussion}

\subsection{Failure-Mode Audit}

We audit five failure modes, each following the structure
\emph{assumption $\to$ violation $\to$ consequence $\to$ mitigation}:
(FM1) non-stationarity ($\times 1.23$ CI widening);
(FM2) independence violations ($\times 1.02$);
(FM3) informative censoring in LOS data ($\times 1.04$);
(FM4) heavy-tailed LOS ($\times 1.14$);
(FM5) Goodhart's Law---staff reacting to forecasts alters observed
rates ($\times 1.05$).
The five penalties compound to a \textbf{$\times 2.27$ CI widening},
stretching the crowding probability interval to 18--94\%.

\subsection{Limitations}

The discharge model ignores the \emph{boarding effect}---LOS
increasing with occupancy. The Poisson arrival assumption neglects
autocorrelation that a Hawkes process would capture. The windowed
model discards historical context that a changepoint prior could
retain. Finally, validation on real (de-identified) data is needed
before any clinical deployment.

\subsection{Ethical Considerations}

Three safeguards are embedded in BUICU. First, only synthetic data is
used. Second, every forecast is a distribution, never a point
estimate. Third, failure modes are front-page content on the website,
not a buried appendix. The Goodhart failure mode deserves particular
emphasis: BUICU must never be used for autonomous staffing decisions
without explicit human oversight.

% ── 5. Conclusion ───────────────────────────────────────────────────
\section{Conclusion}

BUICU demonstrates that conjugate Bayesian inference, combined with
Monte Carlo simulation and systematic failure-mode analysis, can
produce honest, calibrated crowding forecasts. The central practical
finding---that bed capacity, not model refinement, is the dominant
lever---reframes the problem from a statistical one to an operational
one. A probabilistic forecast that discloses its own limitations is
more useful than a deterministic prediction that projects false
confidence. The interactive deployment at
\url{https://buicu-cs109.streamlit.app} makes this argument tangible.

\section{Mathematical Details}\label{app:math}

\subsection{Conjugate Update Derivation}

The Gamma--Poisson conjugacy yields:
\[
    \lambda \mid k_{1:T} \;\sim\; \mathrm{Gamma}\!\left(
        \alpha_0 + \textstyle\sum_{t=1}^T k_t,\;\;
        \beta_0 + T
    \right).
\]
The posterior mean $\alpha/\beta$ converges to the sample mean; the
variance $\alpha/\beta^2$ shrinks as $O(1/T)$.

\subsection{KL Divergence Between Successive Posteriors}

For two Gamma distributions $P_{\text{new}}$ and $P_{\text{old}}$:
\begin{align}
    &\mathrm{KL}(P_\text{new} \| P_\text{old})
    = (\alpha_n - \alpha_o)\psi(\alpha_n) \nonumber\\
    &\quad- \ln\Gamma(\alpha_n) + \ln\Gamma(\alpha_o) \nonumber\\
    &\quad+ \alpha_o(\ln\beta_n - \ln\beta_o)
    + \alpha_n\frac{\beta_o - \beta_n}{\beta_n}
\end{align}

\subsection{Variance Decomposition}

For $N \mid \lambda \sim \mathrm{Poisson}(\lambda \Delta t)$:
\begin{align}
    E[\mathrm{Var}[N \mid \lambda]] &= \Delta t \cdot \alpha/\beta
    \quad\text{(aleatoric)} \\
    \mathrm{Var}[E[N \mid \lambda]] &= \Delta t^2 \cdot \alpha/\beta^2
    \quad\text{(epistemic)}
\end{align}

\section{Failure-Mode Details}\label{app:failure}

\noindent\textbf{FM1 --- Non-stationarity.}
Surge/normal ratio of 1.76 exceeds 1.3 threshold.
Mitigation: windowed model. CI widening: $\times 1.23$.

\medskip\noindent\textbf{FM2 --- Independence violations.}
Lag-1 autocorrelation $r = 0.08$; LOS increases during crowding (boarding).
Mitigation: Hawkes process. CI widening: $\times 1.02$.

\medskip\noindent\textbf{FM3 --- Data quality.}
2.1\% missing discharge timestamps (informative right-censoring).
Mitigation: Kaplan--Meier. CI widening: $\times 1.04$.

\medskip\noindent\textbf{FM4 --- Heavy tails.}
LOS excess kurtosis 5.8.
Mitigation: Generalized Pareto tail fit. CI widening: $\times 1.14$.

\medskip\noindent\textbf{FM5 --- Goodhart's Law.}
Forecasts alter staff behaviour, biasing observed rates.
Cannot be empirically detected; flagged unconditionally.
CI widening: $\times 1.05$.
